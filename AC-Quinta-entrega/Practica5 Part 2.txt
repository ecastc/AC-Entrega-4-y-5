LineToLineBlue PROC

	beginwhileX: 
		cmp countX,8
		je endwhileX
		
		cmp paridad,0
		je	FilaImpar
		jmp FilaPar


FilaImpar:
		
		cmp colorImpar,0
		je ColorWhiteImpar
		jmp ColorBlueImpar

		ColorWhiteImpar:
			beginwhileYwhiteImpar: 
				cmp countY,2   
				je endwhileY
					mov colorImpar,1
					mov	eax,white + (White * 16)
					call	SetTextColor
					mWriteAt valX,valY,"    "
					call	Crlf
					inc	  valY
					inc	  countY
				jmp beginwhileYwhiteImpar     ; repeat the loop 

		ColorBlueImpar:
			beginwhileYblueImpar: 
				cmp countY,2   
				je endwhileY
					mov colorImpar,0
					mov	eax,white + (Blue * 16)
					call	SetTextColor
					mWriteAt valX,valY,"    "
					call	Crlf
					inc	  valY
					inc	  countY
				jmp beginwhileYblueImpar     ; repeat the loop 

FilaPar:

		cmp colorPar,0
		je ColorWhitePar
		jmp ColorBluePar

		ColorWhitePar:
			beginwhileYwhitePar: 
				cmp countY,2   
				je endwhileY
					mov colorPar,1
					mov	eax,white + (White * 16)
					call	SetTextColor
					mWriteAt valX,valY,"    "
					call	Crlf
					inc	  valY
					inc	  countY
				jmp beginwhileYwhitePar     ; repeat the loop 

		ColorBluePar:
			beginwhileYbluePar: 
				cmp countY,2   
				je endwhileY
					mov colorPar,0
					mov	eax,white + (Blue * 16)
					call	SetTextColor
					mWriteAt valX,valY,"    "
					call	Crlf
					inc	  valY
					inc	  countY
				jmp beginwhileYbluePar     ; repeat the loop 


		endwhileY:
			cmp fila,8
				jl FILA1
				je FILA1
					cmp fila,16
					jl FILA2
					je FILA2
						cmp fila,24
						jl FILA3
						je FILA3
							cmp fila,32
							jl FILA4
							je FILA4
								cmp fila,40
								jl FILA5
								je FILA5
									cmp fila,48
									jl FILA6
									je FILA6
										cmp fila,56
										jl FILA7
										je FILA7
										jg FILA8
									
			
FILA1:	
	mov valY,2
	jmp L2

FILA2:	 
	mov valY,4
	jmp L2

FILA3:	 
	mov valY,6
	jmp L2

FILA4:	 
	mov valY,8
	jmp L2

FILA5:	 
	mov valY,10
	jmp L2

FILA6:	 
	mov valY,12
	jmp L2

FILA7:	 
	mov valY,14
	jmp L2

FILA8:	 
	mov valY,16
	jmp L2


			L2:

			
			mov countY,0
			add valX,4
			inc	countX
			inc fila
		jmp beginwhileX     ; repeat the loop 
	endwhileX:

	ret
LineToLineBlue ENDP
